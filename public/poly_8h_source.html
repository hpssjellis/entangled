130 // a * b (mods q)
131 static inline poly_coeff_t poly_coeff_mul(poly_coeff_t const a, poly_coeff_t const b) {
132 #if defined(MAM_POLY_MRED_BINARY)
133  return poly_coeff_mredd((poly_dcoeff_t)a * b);
134 #else
135  poly_dcoeff_t prod = (poly_dcoeff_t)a * b;
136 
137  prod = MAM_MODS(prod, ((trint18_t)MAM_POLY_Q * (trint18_t)MAM_POLY_Q), MAM_POLY_Q);
138 
139  return prod;
140 #endif
141 }
