43 #define JSON_CHECK_ERROR(OBJ, ITEM, LOGGER) \
44  do { \
45  if (OBJ == NULL) { \
46  log_error(LOGGER, "[%s:%d] %s\n", __func__, __LINE__, STR_CCLIENT_JSON_PARSE); \
47  cJSON_Delete(OBJ); \
48  return RC_CCLIENT_JSON_PARSE; \
49  } \
50  ITEM = cJSON_GetObjectItemCaseSensitive(OBJ, "error"); \
51  if (cJSON_IsString(ITEM) && (ITEM->valuestring != NULL)) { \
52  log_error(LOGGER, "[%s:%d] %s %s\n", __func__, __LINE__, STR_CCLIENT_RES_ERROR, ITEM->valuestring); \
53  cJSON_Delete(OBJ); \
54  return RC_CCLIENT_RES_ERROR; \
55  } \
56  ITEM = cJSON_GetObjectItemCaseSensitive(OBJ, "exception"); \
57  if (cJSON_IsString(ITEM) && (ITEM->valuestring != NULL)) { \
58  log_error(LOGGER, "[%s:%d] %s %s\n", __func__, __LINE__, STR_CCLIENT_RES_ERROR, ITEM->valuestring); \
59  cJSON_Delete(OBJ); \
60  return RC_CCLIENT_RES_ERROR; \
61  } \
62  } while (0);
