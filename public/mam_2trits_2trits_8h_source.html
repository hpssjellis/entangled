113 static inline bool trits_is_overlapped(trits_t x, trits_t y) {
114  trit_t *x_first = x.p + x.d;
115  trit_t *x_last = x.p + x.n;
116  trit_t *y_first = y.p + y.d;
117  trit_t *y_last = y.p + y.n;
118  return (x_first < y_last) && (y_first < x_last);
119 }
