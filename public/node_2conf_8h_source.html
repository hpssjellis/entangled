39  // hash. Difficulty can be described as 3^mwm
40  uint8_t mwm;
41  // URIs of neighbouring nodes, separated by a space
42  char* neighbors;
43  // Probability of sending a milestone transaction when the node looks for a
44  // random transaction to send to a neighbor. Value must be in [0,1]
45  double p_send_milestone;
46  // Size of the tips cache
47  size_t tips_cache_size;
48  // The number of entries to keep in the network cache

50  // Size of the requester queue
51  size_t requester_queue_size;
52  // Path of the tangle database file

54  // The address of the coordinator encoded in bytes
55  byte_t coordinator_address[HASH_LENGTH_BYTE];
56  // The address to bind the TCP server socket to
57  char neighboring_address[16];
58  // The TCP receiver port
59  uint16_t neighboring_port;

62  bool auto_tethering_enabled;
63  // The maximum number of neighbors allowed to be connected
64  size_t max_neighbors;
