84 static inline trit_t flex_trits_at(flex_trit_t const *const flex_trits, size_t const len, size_t index) {
85  // Bounds checking
86  if (index >= len) {
87  return 0;
88  }
89 #if defined(FLEX_TRIT_ENCODING_1_TRIT_PER_BYTE)
90  // Straight forward 1 trit per byte
91  return flex_trits[index];
92 #elif defined(FLEX_TRIT_ENCODING_3_TRITS_PER_BYTE)
93  return get_trit_at((tryte_t *)flex_trits, NUM_FLEX_TRITS_FOR_TRITS(len), index);
94 #elif defined(FLEX_TRIT_ENCODING_4_TRITS_PER_BYTE)
95  // Find out the position of the trit in the byte
96  uint8_t mshift = (index & 3) << 1U;
97  // Calculate the shift to sign extend the result
98  uint8_t tshift = 6U - mshift;
99  // Find out the index of the byte in the array
100  index = index >> 2U;
101  // Get the trit and sign extend it
102  return (flex_trit_t)(flex_trits[index] << tshift) >> 6U;
103 #elif defined(FLEX_TRIT_ENCODING_5_TRITS_PER_BYTE)
104  trit_t trits[5];
105  // Find out the index of the trit in the byte
106  uint8_t tindex = index % 5U;
107  // Find out the index of the byte in the array
108  index = index / 5U;
109  byte_to_trits(*(flex_trits + index), trits, 5);
110  return trits[tindex];
111 #endif
112 }

119 static inline uint8_t flex_trits_set_at(flex_trit_t *const flex_trits, size_t const len, size_t index, trit_t trit) {
120  // Bounds checking
121  if (index >= len) {
122  return 0;
123  }
124 #if defined(FLEX_TRIT_ENCODING_1_TRIT_PER_BYTE)
125  // Straight forward 1 trit per byte
126  flex_trits[index] = trit;
127 #elif defined(FLEX_TRIT_ENCODING_3_TRITS_PER_BYTE)
128  set_trit_at(flex_trits, NUM_FLEX_TRITS_FOR_TRITS(len), index, trit);
129 #elif defined(FLEX_TRIT_ENCODING_4_TRITS_PER_BYTE)
130  // Calculate the final position of the trit in the byte
131  uint8_t shift = (index & 3) << 1U;
132  // Position the trit at its place in the byte
133  trit = (trit & 0x03) << shift;
134  // Create a mask to reset the target trit
135  uint8_t mask = ~(0x03 << shift);
136  // Find out the index of the byte in the array
137  index = index >> 2U;
138  // bitblit the trit in place
139  flex_trits[index] = (flex_trits[index] & mask) | trit;
140 #elif defined(FLEX_TRIT_ENCODING_5_TRITS_PER_BYTE)
141  trit_t trits[5];
142  // Find out the index of the trit in the byte
143  uint8_t tindex = index % 5U;
144  // Find out the index of the byte in the array
145  index = index / 5U;
146  byte_to_trits(*(flex_trits + index), trits, 5);
147  trits[tindex] = trit;
148  flex_trits[index] = trits_to_byte(trits, 5);
149 #endif
150  return 1;
151 }
